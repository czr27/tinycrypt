Introduction

ChaCha is a stream cipher with support for 128 and 256-bit keys. It was designed by Daniel Bernstein and published in 2008. Because of an add-rotate-xor (ARX) construction, this makes it suitable for a wide range of architectures. It prevents leakage of information through side channel analysis, has a simple and fast key setup, and provides good overall performance. Google have published details of their own implementation in RFC 8439. They will use ChaCha for symmetric encryption along with Poly1305 for authentication.

Macros, data types

The following macros and data types are used are used throughout source.

#define R(v,n)(((v)>>(n))|((v)<<(32-(n))))
#define F(n)for(i=0;i<n;i++)
#define X(a,b)(t)=(a),(a)=(b),(b)=(t)

typedef unsigned char B;
typedef unsigned int W;
typedef unsigned long long Q;

typedef union _chacha_ctx_t {
    B b[64];
    W w[16];
    Q q[8];
} chacha_ctx;

ChaCha core

The variant shown here is ChaCha20. There is no difference between the core used by Google and the original. Of course, what you see here is optimized for size.

void chacha20_core(void *input, void *output) {
    W a,b,c,d,i,t,r,*s=input,*x=output;
    W v[8]={0xC840,0xD951,0xEA62,0xFB73,
            0xFA50,0xCB61,0xD872,0xE943};
            
    // load state into local buffer
    F(16)x[i]=s[i];
    
    // for 80 rounds
    F(80) {
      d=v[i%8];
      a=(d&15);b=(d>>4&15);
      c=(d>>8&15);d>>=12;
      
      for(r=0x19181410;r;r>>=8)
        x[a]+=x[b],
        x[d]=R(x[d]^x[a],(r&255)),
        // swap
        X(a,c),X(b,d);
    }
    // add state to local buffer
    F(16)x[i]+=s[i];
}

Initialization

The main changes made by Google are using a 256-bit key by default, a 32-bit counter instead of 64-bits and a 96-bit nonce instead of 64-bits. The source shown here requires 256-bit keys, but the original length of counter and nonce is still supported.

void chacha20_setkey(chacha_ctx *c, void *key, void *nonce) {
    W   *k=(W*)key, *n=(W*)nonce;
    int i;
    
    c->w[0]=0x61707865; c->w[1]=0x3320646E;
    c->w[2]=0x79622D32; c->w[3]=0x6B206574;
    // copy 256-bit key
    F(8)c->w[i+4] = k[i];
  #ifdef GX
    // set 32-bit counter
    c->w[12] = 1;
    // copy 96-bit nonce
    F(3)c->w[i+13] = n[i];
  #else
    // set 64-bit counter
    c->w[12] = 0; c->w[13] = 0;
    // set 64-bit nonce
    F(2)c->w[i+14] = n[i];
  #endif
}

Encryption/Decryption

Because Google only use a 32-bit counter, there is a slight change in updating the counter after generating a stream.

void chacha20_encrypt(chacha_ctx *ctx, void *in, W len) {
    B c[64],*p=in;
    W i,r;

    // if we have data to encrypt/decrypt
    if(len) {
      while(len) {
        // permute state
        chacha20_core(ctx, c);
        // increase counter
      #ifdef GX
        ctx->w[12]++;
      #else
        ctx->q[6]++;
      #endif
        // encrypt 64 bytes or whatever is remaining
        r = (len > 64) ? 64 : len;
        // xor plaintext with ciphertext
        F(r)p[i] ^= c[i];
        // decrease length, advance buffer
        len -= r;
        p   += r;
      }
    }
}

x86 assembly

This can be used to initialize state, and perform encryption or decryption using a single call. See chacha.c for more details.

; -----------------------------------------------
; ChaCha20 stream cipher in x86 assembly
;
; size: 187 bytes for 256-bit key, 96-bit nonce, 32-bit counter
;       189 bytes for 256-bit key, 64-bit nonce, 64-bit counter
;
; global calls use cdecl convention
;
; -----------------------------------------------

    bits 32
 
    %ifndef BIN
      global chacha
      global _chacha
    %endif
    
    ; ------------------------------------
chacha:
_chacha:
    pushad
    lea    esi, [esp+32+4]
    lodsd
    xchg   ecx, eax          ; ecx = len
    lodsd
    xchg   ebx, eax          ; ebx = input or key+nonce
    lodsd
    test   ecx, ecx
    jnz    L0   

    xchg   eax, edi
    mov    esi, ebx
    ; copy "expand 32-byte k" into state
    mov    eax, 0x61707865
    stosd
    mov    eax, 0x3320646E
    stosd
    mov    eax, 0x79622D32
    stosd
    mov    eax, 0x6B206574
    stosd
    ; copy 256-bit key
    mov    cl, 8
    rep    movsd
  %ifdef GX
    ; set 32-bit counter
    push   1
    pop    eax
    stosd                  ; s[12] = 1
    ; F(3)s[i+13] = k[i+8];
    movsd
    movsd
    movsd
  %else
    xor    eax, eax
    stosd                  ; s[12] = 0
    stosd                  ; s[13] = 0
    ; F(2)c->w[i+14] = n[i];
    movsd
    movsd
  %endif
    popad
    ret 
L0:    
    xchg   esi, eax          ; esi = state
    ; perform encryption/decryption
    pushad
    pushad
    mov    edi, esp          ; edi = c
L1:
    xor    eax, eax
    jecxz  L7                ; exit if len==0
    ; permute
    ; esi = state
    ; edi = out
    pushad
    pushad
    ; memcpy(x, s, 64)
    push   64
    pop    ecx
    rep    movsb
    pop    edi
    push   edi
    ; i = 0
    xor    eax, eax
L2:
    push   eax
    call   L3
    dw     040c8H, 051d9H, 062eaH, 073fbH
    dw     050faH, 061cbH, 072d8H, 043e9H
L3:
    pop    esi
    and    al, 7 
    lea    esi, [esi+eax*2]
    
    lodsb
    aam    16
    movzx  edx, al    
    movzx  ebp, ah   
    
    lodsb
    aam    16
    movzx  ebx, ah
    movzx  eax, al
    
    ; for (r=0x7080C10;r;r>>=8)
    mov    ecx, 0x7080C10 ; load rotation values
L4:
    ; x[a] += x[b]
    mov    esi, [edi+ebx*4]
    add    [edi+eax*4], esi
    
    ; x[d] = R(x[d] ^ x[a], (r & 255))
    mov    esi, [edi+ebp*4]    
    xor    esi, [edi+eax*4]
    rol    esi, cl
    mov    [edi+ebp*4], esi
    
    ; X(a, c); X(b, d);
    xchg   eax, edx
    xchg   ebx, ebp 
    
    ; r >>= 8
    shr    ecx, 8       ; shift until done 
    jnz    L4
    
    pop    eax
    inc    eax
    cmp    al, 80
    jnz    L2
    
    popad
    
    ; F(16) x[i] += s[i];
    mov    cl, 16
L5:
    lodsd
    add    [edi], eax
    scasd
    loop   L5
  %ifdef GX
    ; s[12]++;
    inc    dword[esi-4*4]
  %else
    stc
    ; ((Q*)&s)[6]++
    adc    dword[esi-4*4], ecx
    adc    dword[esi-3*4], ecx
  %endif
    popad
L6:
    mov    dl, [edi+eax]
    xor    [ebx], dl
    inc    ebx
    inc    eax
    cmp    al, 64
    loopne L6
    jmp    L1
L7:
    popad
    popad   
    popad
    ret
